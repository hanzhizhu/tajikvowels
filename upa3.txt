آونــدهـايــى بـرضـد دگـرگــونــى دبـيــره ى پــارســى-عـربــى و پـاسـخ هـايــى بــدانـهـا:

۱. چـه كـسـانــى خــواهـنـد تـوانـسـت آن هـمـه كـتـابـهـا ، چـكـامـه هـا ، نـامـه هـا ، و ديـگـر نــوشـتـه هـاى پـارســى را بـه دبـيــره ى لـاتـيـن در آورنــد؟ بـازنــويـســى هـمـه ى نــوشـتـه هـاى يـك جـامـعـه از گـسـتـرده تـريـن بـرنـامـه هـاى تـاريـخ انـسـان خــواهـد بــود . آيـا نـگـارشـهـاى خــوشـنــويـسـان بـر پـيـكـر سـاخـتـمـانـهـاى تـاريـخــى را نـيــز بـايـد بـاز نــوشـت ؟
پـاسـخ: 
رونـد دگـرگــونــى الـفـبـا از پــارســى-عـربــى بـه يـك الـفـبـاى واجــى بـر پـايـه ى لـاتـيـن ، مـانـنـد پـارســى جـهـانــى ، بـايـد خـرد خـرد و روشـمـنــدانـه انـجـام شـود . بـراى چــونـيـن دگـرديـسـيــى نـيـاز بـه يـك بـرنـامـه ريــزى سـنـجـيــده اسـت . ايـن بـرنـامـه بـايـد در چـنـد گـام و دوره رخ بــدهـد:

دوره ى آغـازيـن - در ايــن دوره ، دبـيــره ى نـو لـاتـيـن در آمــوزشـگـاهـهـا بـه عـنــوان يـك الـفـبـاى سـنـجـه ى واجــى بـراى نـگـارش پـارســى ، جـزء بـرنـامـه ى آمــوزشــى ، شـنـاسـانــده مـيـشـود . الـفـبـاى سـاده يــى چــون ايــن را مـيـتـوان آسـان ، بـيــدرد سـر ، و در كـمـتـريـن زمـان بـه دانـش انــدوزان پـارســى زبـان آمــوزانـد . در فـراگــويـش و يـادگـيــرى واژه هـاى دشــوار ، دبـيــره ى واجــى ، بــديـشـان يـارى خــواهـد كـرد . بـه دانـش انــدوزانــى كـه پـارســى يـاد نـدارنـد ، خــوانــدن نــوشـتـه هـا بـه دبـيــره ى نـو لـاتـيـن ، بـنـد بـزرگ دبـيــره ى پـيـچـيــده ى پــارســى-عـربــى را كـه بـر سـر راه فـراگـيــريــى زبـان پـارســى جـاى گــرفـتـه ، بـر خــواهـد داشـت . تـنـهـا كـتـابـهـايــى كـه در ايـن زمـان بـايـد بـه دبـيــره ى نـو بـاز نــوشـتـه شـونـد ، كـتـابـهـاى آمــوزش زبـان پـارســى ، و اثــرهـاى بـزرگ اديـبـانــى چــون شـاهـنـامـه ى فــردوســى ، مـثـنـوى بـلـخــى ، و ديـگـران هـسـتـنـد .
دوره ى هـمــزيـسـتــى - در ايــن زمـان ، بـيـش از پـيـش ، كـتـابـهـا ، نـشـانـه هـا ، نــوشـتـه هـاى كـهـن و تـازه بـه دبـيــره ى نـو بـاز نــوشـتـه مـيـشـونـد . بـازنــويـســى مـتـنـهـاى پــارســى-عـربــى بـه دبـيــره ى نـو ، مـيـتـوانـد بـه صــورت خــودكـار و بـا دخـالـت هـرچـه كـمـتـر از ســوى كـاربـران انـجـام گـيــرد . ازيـنــرو ، كــوشـشـهـاو رنـجـهـاى اسـتـادان بـه انــدازه ى زيـادى كـاهـش خــواهـد يـافـت . تـنـهـا تـلـاش بـزرگ ، وارســى و بـازبـيـنــى مـتـنـهـا از ســوى زبـانـشـنـاسـان ، بـراى يـافـتـن لـغــزشـهـايــى در كـار بـازنــويـســى خــواهـد بــود . در ايــن دوره ، بـيـشـتـر پـارســى زبـانـان بـا هـرد و دبـيــره ى پــارســى-عـربــى و نـو لـاتـيـن آشـنـا خــواهـنـد بــود ، و آشـكـارا بـرتـرى دبـيــره ى واجــى لـاتـيـن را بـراى خــود خــواهـنـد ديـد .
دوره ى واپـسـيـن - در ايــن زمـان ، الـفـبـاى نـو بـه عـنــوان يـك ابــزار كـارا و عـمـلــى بـراى نـگـاشـتـن هـر جــور كـتـاب ، مـدرك ، و نــوشـتـه ى الـكـتــرنـيـكــى و غـيــره پـابـرجـا شـده اسـت . ايـن دوره ى جـايـگـزيـنــى الـفـبـاى نـو بـه جـاى الـفـبـاى كـهـن اسـت . بـا وجــود دگـرگــونــى دبـيــره ، كـار خــوشـنــويـســى بـه دبـيــره ى كـهـن پـيــوسـتـه دنـبـال خــواهـد شـد و خــوشـنــويـســى بـه دبـيــره ى نـو لـاتـيـن نـيــز بـه ايــن هـنـر افــزوده خــواهـد شـد .

۲. فـرهـنـگ كـهـن پـارســى چـه خــواهـد شـد؟ آيـا از دسـت نـخــواهـد رفـت ؟

پـاسـخ: 
نـخـسـت بـايـد د و زمـيـنـه ى بـسـيـار مـهـم ، يـكــى زبـان و ديـگـرى دبـيــره يــى كـه زبـان بــدان نـگـاشـتـه مـيـشـود را از يـكــديـگـر بـاز شـنـاخـت . زبـان و فـرهـنـگ بـه يـكــديـگـر هـمـبـسـتـگــى نـزديـك دارنـد . مـارتـيـن هـايــدگــر ، فـيـلـســوف بـزرگ آلـمـانــى در بـاره ى ايـن هـمـبـسـتـگــى مـيـگــويـد : 

"زبـان سـراى هـسـتــى اسـت".

در ايــن گـفـتـه مـيـتـوان فـرهـنـگ را جـايـگـزيـن واژه ى هـسـتــى كـرد . دبـيــره يـا الـفـبـا تـنـهـا ابــزارى اسـت بـراى خــوانــدن و نــوشـتـن . دبـيــره زبـان نـيـسـت ، و هـيـچـگــونـه وابـسـتـگـيــى بـه زبـان و فـرهـنـگ نـدارد . هـمـانـنـد جـامـه يــى بـر تـن اسـت . تـن مـانـنـد زبـان و دبـيــره هـمـچــون جـامـه اسـت . در تـن هـرگــز بـا پــوشـيــدن جـامـه هـاى گــونـاگــون دگـرگــونـيــى رخ نـخــواهـد داد . هـر زبـانــى را مـيـتـوان بـه هـر الـفـبـايــى نــوشـت . خــوش پــوشـتـريـن جـامـه هـا آنـهـايــى هـسـتـنـد كـه بـراى انــدازه هـاى ويــژه ى آن تـن دوخـتـه شـده بـاشـنـد . جـامـه هـايــى كـه بـرازنــده ى آن تـن بـاشـنـد . بـرازنــده تـريـن دبـيــره بـراى يـك زبـان آن اسـت كـه بـراى ويــژگـيـهـا و آواهـاى آن زبـان سـاخـتـه شـده بـاشـد .

دبـيــره ى پــارســى-عـربــى ، ارزش فـرهـنـگــى كـهـتـر يـا بـرتـرى نـسـبـت بـه دبـيــره ى پـارســى جـهـانــى نـدارد . فـرقـشـان ايـنـجـاسـت كـه دبـيــره ى پـارســى جـهـانــى بـرازنــده ى تـن زبـان پـارســى اسـت ولــى پــارســى-عـربــى نـيـسـت . پــارســى-عـربــى هـمـان دبـيــره ى پـيـچ درپـيـچ عـربـــى اسـت كـه بــدان چـهـار نــويـسـه ى ويــژه ى پـارســى افــزوده شـده اسـت . كـدام دبـيــره شـايـسـتـه ى زبـان و فـرهـنـگ پـارســى اسـت ؟ آن كـه سـنـگـيـن اسـت و خــوانــدن و نــوشـتـن را دشــوار مـيـكـنـد يـا آن كـه بـرازنــده و آسـان اسـت ؟ فـرجـام ايــن اسـت كـه گـزيـنـش يـك دبـيــره ى واجــى بـر پـايـه ى لـاتـيـن ، بـراى زبـان و فـرهـنـگ پـارســى شـايـسـتـه تـر اسـت . چــون نـه تـنـهـا بــاعــث  از مـيـان رفـتـن آن فـرهـنـگ كـهـن نـخــواهـد شـد ، بـلـكـه بـا گـشــودن راه دسـتــرســى آسـانـتـر بـه آن از ســوى پـارســى زبـانـان و نـو آمــوزان ، بــاعــث شـكــوفـايــى بـيـش از پـيـش فـرهـنـگ پـارســى و آشـنـايــى جـهـانـيـان بــدان خــواهـد گـشـت .

۳. دبـيــره ى نـو بـه ادب پـارســى و بـازمـانــده ى فـرهـنـگــى مـا آسـيـب خــواهـد رسـانـيـد . ديـگـر هـيـچـكـس نـخــواهـد تـوانـسـت نـگـاشـتـه هـاى كـهـن بـر ديــوارهـاى سـاخـتـمـانـهـاى تـاريـخــى را بـخــوانـد .

پـاسـخ: 
آونـد بـالـا امـكـان دارد كـه نـخـسـت گـيــرا و درسـت بـه چـشـم بـيـايـد . بــويــژه ، بـراى كـسـانــى كـه آشـنـايــى بـا دبـيــره ى پــارســى-عـربــى و دشــواريـهـاى بـيـش از انــدازه ى آن نـدارنـد . حـقـيـقـت ايـنـجـاسـت كـه پـيـچـيــدگـيـهـاى دبـيــره ى كـنــونــى از بـررســى و آگـاهـش پـارســى زبـانـان در ادب زبـان خــودشـان پـيـشـگـيــرى كـرده اسـت . نـهـاد بــى در و پـيـكـر دبـيــره ى پــارســى-عـربــى ، بـررســى در ادب پـارســى را تـنـهـا در اخـتـيـار سـرامـدان گــذاشـتـه اسـت . دبـيــره يــى واجــى بـر پـايـه ى لـاتـيـن ، مـانـنـد پـارســى جـهـانــى ، ادب پـارســى را از زنــدان آزاد كـرده و در دسـتــرســى هـمـگـان خــواهـد گــذاشـت . بـاز ايــن كـه ، گـزيـنـش دبـيــره ى پـارســى جـهـانــى ، ادب و فـرهـنـگ پـارســى را از جـدايــى رهـايــى بـخـشـيــده ، و قــدردانــى شـايـان آنـان را امـكـانـپــذيــر مـيـسـازد . نـبـشـتـه هـاى ديــوارهـا را نـمـيـتـوان بـاز نــوشـت ، ولــى راهـهـاى بـسـيـارى بـراى بـازنــويـســى آنـان بـر روى صـفـحـه هـا و كـتـابـهـا ، بـراى خــوانـا بــودن ، وجــود دارد . بـراى نـمــونـه نـبـشـتـه هـايــى بـه دبـيــره ى مـيـخــى در تـخـت جـمـشـيـد و بـهـيـسـتـان در ايــران هـسـتـنـد كـه بـراى تــوده ى مـردم خــوانـا نـيـسـتـنـد . ولــى ايــن از بـازنــويـســى و تـرجـمـه شـدن چـنــديـن بـاره ، و در دسـتــرس هـمـگــى گــذاشـتـنـشـان ، پـيـشـگـيــرى نـكـرده اسـت . بـرنـامـه ى عـلـمــى بـازنــويـســى نـبـشـتـه هـاى مـيـخــى بـه دسـت دانـشـمـنــدان ، بـه گـسـتـرش و افــزايـش دانـش دربـاره ى آنـان بـسـيـار يـارى كـرده اسـت . خــوانــدن نـبـشـتـه هـاى كـهـن بـر ديــوارهـا و كـتـابـهـا ارزش فـراوانــى دارد ، ولــى نـقـشــى بـسـيـار كــوچـك بـراى الـفـبـاى يـك زبـان اسـت . بـايـد هـمـيـشـه بـه خــود يـاد آورى كـنـيـم كـه نـقـش بـنـيـادى و پـايـه يــى يـك الـفـبـا ، آسـان سـازى ارتـبـاط مـيـان مـردم ، و از مـيـان بـرداشـتـن بـنــدهـا بـر سـر راه خــوانــدن و نــوشـتـن اسـت .

۴. روش نـگـاشـتـن زبـان ژاپـنــى ، بـا ۴ دبـيــره ى گــونـاگــون ، از پـيـچـيــده تـريـن در گـيـتــى اسـت . بـا ايــن حـال ، درصـد بـاسـوادان آن كـشــور از تـركـيـه و ويـتـنـام ، كـه الـفـبـاى لـاتـيـن را بـر گـزيــده انـد ، بـالـاتـر اسـت .

پـاسـخ: 
زبـان ژاپـنــى ۴ گــونـه دبـيــره دارد بـه نـامـهـاى هـيــراگـانـا ، كـاتـاكـانـا ، كـانـجــى ، و رمـاجــى . هـيــراگـانـا و كـاتـاكـانـا هـجـانـگـاريـنـد و بـرازنــده ى آن زبـانـنـد . كـانـجــى يـك گـروه از نـمـادهـاى چـيـنـيـنـد كـه در ژاپـنــى بـه كـار مـيــرونـد . رمـاجــى دبـيــره ايـسـت لـاتـيـنــى بـراى آوانــويـســى زبـان ژاپـنــى . از مـيـان ايــن ۴ دبـيــره ، رمـاجــى ويــژه ى نــوشـتـن شـمـاره هـا و كــوتـه نــوشـتـهـا اسـت . در ضـمـن بـراى نــوشـتـن فـرهـنـگـهـا ، كـتـابـهـاى درســى ، و عـبـارتـنـامـه هـاى بـيـگـانـگـانــى كـه دوسـتـار فـراگـيــريــى زبـان ژاپـنــى انـد نـيــز بـه كـار مـيــرود . هـنـگـام كـاربـرد از رايـانـه ، بـيـشـتـر مـردم ، چـه ژاپـنــى و چـه بـيـگـانـه ، رمـاجــى را بـه كـار مـيـبـرنـد . آشـكـارا اسـت كـه نـهـاد پـيـچـيــده و نـاعـمـلــى دبـيــره هـاى غـيــر لـاتـيـنــى ژاپـنــى نـيـاز بـه يـك دبـيــره ى لـاتـيـن را مـوجـب شـده اسـت . ايـن درسـت چـيــزيـسـت كـه بـراى زبـان پـارســى بـايـد انـجـام شـود . پـارســى بـه يـك دبـيــره ى سـاده و آسـان ، بـر پـايـه ى لـاتـيـن ، در ايـن دوره ى رايـانـه و ايـنـتــرنـت نـيـازمـنـد اسـت . 

هـر جــور مـقـايـسـه ى ژاپـن بـا تـركـيـه يـا ويـتـنـام خـنــده آور اسـت . ابـرنـيــرويــى اقـتـصـادى چــون ژاپـن ، تـوانـايــى آن را دارد كـه مـردمـانـش را بـه هـر دبـيــره يــى بـاسـواد كـنـد . ولــى بـراى كـشــورهـايــى چــون تـركـيـه ، ويـتـنـام يـا ايــران ، بـالـا بـردن پـايـه ى آمــوزش و پـرورش شـهــرونــدانـشـان بـه هـزيـنـه هـايــى بـس هـنـگـفـت و نـايـاب نـيـاز دارد . چــرا دارايــى و تـوانـايــى خــود را در راه نـگـهــدارى دبـيــره يــى بـيـهــوده و ازكـار افـتـاده بـه كـار بـبـريـم ؟ يـك دبـيــره ى واجــى لـاتـيـن بـه بـالـا بـردن پـايـه ى آمــوزشــى بـا كـمـتـريـن هـزيـنـه و در كــوتـاهـتـريـن زمـان ، يـاورى خــواهـد كـرد .

پـيـچ درپـيـچــى دبـيــره هـاى ژاپـنــى اثــر مـنـفــى بـر روى فـرهـنـگـشـان گــذاشـتـه اسـت . دانـش آمــوزان ژاپـنــى تـا سـن ۱۲ سـالـگــى تـوانـايــى خــوانـدن روزنـامـه را نـدارنـد . بـه چـم ايــن اسـت كـه ايـشـان بـايـد ۶ سـال آزگـار دانـش خــوانــدن نــويـسـه هـاى گــونـاگــون را بـيـنــدوزنـد تـا تـوانـايــى خــوانــدن روزنـامـه را در يـابـنـد . بـنـابـر ايـن خــوانــدن كـتـابـهـاى ادبــى تـا رسـيــدگــى سـنــى ايـشـان امـكـانـپــذيــر نـخــواهـد بــود . درسـت هـمـيـن دشــوارى در ايــران نـيــز وجــود دارد . پـرسـش ايــن اسـت : چـنـد درصـد مـردمـان بـالـغ ايــرانــى تـوانـايــى خــوانــدن بــى لـغــزش و بــى دشــوارى اثــرهـاى بـزرگـان ادبــى مـا هـمـچــون سـعــدى و فــردوســى را دارنــد؟ انـگـيــزه ى ايــن پـرسـش ايــن نـيـسـت كـه بـبـيـنـيـم چـنـد درصـد مـيـتـوانـنـد مـعـنــى ايــن اثــرهـا را در يـابـنـد . انـگـيــزه ايــن اسـت كـه چـنـد درصـد تـنـهـا تـوانـايــى خــوانــدن واژه هـا را بـيـلـغــزش دارنـد . چــونـيـن درصـد ى بـسـيـار پـايـيـن اسـت . پـرسـشــى ديـگـر پـيـش مـيـايـد . آيـا بـه ســود فـرهـنـگ و ادب پـارســى اسـت كـه بـه دبـيــره يــى نـاخــوانـا و پـيـچـيــده نــوشـتـه شـود ، يـا بـه دبـيــره يــى سـاده و خــوانـا كـه هـمـگــى ، حـتــى آنـان كـه بـا آن زبـان آشـنـايــى نـدارنـد ، بـتـوانـنـد بـه آسـانــى بـخــوانـنــد؟

۵. دبـيــره هـاى تـركــى و ويـتـنـامــى كـه بـر پـايـه ى لـاتـيـن سـاخـتـه شـده انـد ، نــويـسـه هـاى فـراوانــى بـا نـشـانـه هـاى زيــر و زبـرى دارنـد . پـس دشــوارى ايــن نـشـانـه هـا در دبـيــره ى نـو نـيــز مـانـنـد پــارســى-عـربــى ، بـر جـاى خــواهـد مـانـد . 

پـاسـخ:
زبـان پـارســى بـا داشـتـن تـنـهـا ۶ واكـه ، بـراى نـگـاشـتـن بـه دبـيــره ى واجــى لـاتـيـن بـسـيـار بـســزاسـت . بـنـابـر ايـن بـه دسـتـكـارى كـمــى در دبـيــره ى لـاتـيـن نـيـاز دارد . تـنـهـا بـه افــزايـش ۳ نــويـسـه بـا زيــر و زبـر بـه دبـيــره ى لـاتـيـن نـيـاز اسـت ، تـا آن را بـراى پـارســى شـايـنــده بـكـنـد . از مـيـان ايــن ۳ نــويـسـه تـنـهـا نــويـسـه ى زبـردار â ( ا ) فـراوان در مـتـنـهـاى پـارســى رخ مـيــدهـد . نــويـسـه ى š ( ش ) بـسـيـار كـمـتـر و ž ( ژ ) بـسـيـار كـمـيـاب اسـت . پـس در بـيـشـتـر مـتـنـهـاى پـارســى تـنـهـا يـك نــويـسـه ى زبـردار رخ مـيــدهـد . ايــن را بـا دبـيــره ى پــارســى-عـربــى مـقـايـسـه كـنـيـد كـه از مـيـان ۳۳ نــويـسـه تـنـهـا ۱۷ نــويـسـه را مـيـتـوان از ديـگـران بـازشـنـاخـت ، و بـقـيـه ى نــويـسـه هـا داراى خـجـكـهـا و سـركـج هـسـتـنـد . نـشـانـه هـاى واكـه هـاى كــوتـاه ، اگـر نــوشـتـه شـونـد ، بـه شـمـاره ى زيــر و زبـرهـا مـيـفــزايـنـد . ايــن هـمـه نـشـانـه ى افــزوده شـده ، زيــر و زبـر واژه هـا ، نـاخــوانـا بــودن و پـيـچـيــدگــى آنـان را صـدچـنــدان مـيـكـنـد .

۶. انـگـيــزه ىبـنـيـادى ايــرانـيـانــى كـه پـيــروى از دگـرگــونــى دبـيــره مـيـكـنـنـد هـمـان اسـت كـه آل احـمـد در كـتـابـش " غـربــزدگــى " نــوشـتـه اسـت .

پـاسـخ:
دبـيــره ى پـارســى جـهـانــى بـر پـايـه ى دبـيــره لـاتـيـن نـهـاده شـده كـه بــدرسـتــى بـايـد دبـيــره يــى " جـهـانــى " نـامـيــده شـود . درسـت اسـت كـه از ديـد تـاريـخــى دبـيــره ى لـاتـيـن در غـرب پـديــدار گـشـت ، ولــى امــروزه ديـگـر ايــن دبـيــره را غـربــى نـامـيــدن نـادرسـت اسـت . دبـيــره ى لـاتـيـن تـنـهـا مـحــدود بـه جـهـان غـرب نـيـسـت . آن را در آفــريـقـا ، آسـيـا ، و هـمـچــونـيـن در غـرب بـه كـار مـيـبـرنـد . ولــى دبـيــره ى پــارســى-عـربــى را تـنـهـا مـردمـان چـنـد كـشــور آسـيـا ى جـنــوب غـربــى در مـيـابـنـد .

بـسـيـار مـهـم اسـت كـه دبـيــره ى نـو بـر شـالــوده ى دبـيــره ى لـاتـيـن نـهـاده شـود . يـكــى از انـگـيــزه هـاى مـهـم آفـريـنـش دبـيــره ى نــو پـارســى ، گـسـتـرش آمــوزش ايــن زبـان و ادب گـنـجــور آن در مـيـان هـمـه ى جـهـانـيـان اسـت . زبـان بـزرگ و شـكــوهـمـنـد پـارســى در بـنـد دبـيــره ى پـيـچـيــده ، نـاهـنـجـار ، و نـا آشـنـاى پــارســى-عـربــى در آمـده اسـت . دبـيــره ى پـارســى جـهـانــى بـنــدهـا را خــواهـد گـسـسـت ، پـارســى را آزاد كـرده بـه اوج فـر و شـكــوه هـرچـه بـيـشـتـر خــواهـد رسـانـد . كـسـانــى كـه پـارسـيــزبـان نـيـسـتـنـد ولــى دوسـتــدار يـادگـيــرى ايــن زبـان هـسـتـنـد نـيــز خــواهـنـد تـوانـسـت بـه آسـانــى پـارســى را فـرا گــرفـتـه و بـه گـنـجـيـنـه هـاى ادبــى آن دسـتــرســى يـابـنـد . فـرجـام كـار ايــن خــواهـد بــود كـه شــور و جــوش و خـروش تـازه يــى در جـهـان پـيــرامــون آمــوزش زبـان و فـرهـنـگ پـارســى و رشـتـه هـاى پـارســى شـنـاســى و ايــرانـشـنـاســى آغـاز خــواهـد گـشـت . " پـارســى دوسـتــى " جـايـگـزيـن غـربــزدگــى خــواهـد شـد . پـارســى خــود زبـانــى سـاده اسـت . آنـچـه كـه بــاعــث جـلـوگـيــرى فـراگـيــرى و دلـبـسـتـگــى بـه ايــن زبـان شـده اسـت الـفـبـاى پــارســى-عـربــى مـيـبـاشـد .

چــرا الـفـبـاى آوايــى مـوجــود اوسـتـايــى را جـايـگـزيـن الـفـبـاى پــارســى-عـربــى نـكـنـيـم ؟ بـسـيـارى چــونـيـن پـرسـشــى را در مـيـان مـيـگــذارنـد . ايـشـان خــواهـان بـهــره بـردن از آوايــى بــودن ايــن الـفـبـا و پـيـشـگـيــرى از غـربــزده جـلــوه گـر شـدن هـسـتـنـد . اگـرچـه انـگـيــزه ى ايــن گـروه از ديـد مـيـهـن پـرسـتــى قـابـل سـتـايـش اسـت ، ولــى از ديـد كـارايــى عـمـلــى نـيـسـت . دبـيــره ى لـاتـيـن بـراى بـيـشـتـر پـارسـيــزبـانـان و جـهـانـيـان آشـنـاو و شـنـاخـتـه شـده اسـت . بـنـابـر ايـن آمــوزانــدن دبـيــره ى پـارســى جـهـانــى زمـان و كــوشـش كـمـتـرى خــواهـد بـرد . ولــى عـوض كـردن يـك الـفـبـاى كـم آشـنـا بـا يـك الـفـبـاى نـا آشـنـا بـه آمــوزش فـراوان و دگـرگــونــى هـمـه ى ابــزارهـاى چـاپ و رايـانـه هـا نـيـازمـنـد اسـت . بـا گـزيـن دبـيــره ى نـوى بـر پـايـه لـاتـيـن مـيـتـوان از ايــن بـنــدهـا و هـزيـنـه هـا پـرهـيـخـت . در بـاره ى نـنـگ غـربــزدگــى بـايـد گـفـت كـه امــروزه دبـيــره يــى را لـاتـيـن يـا فـرنـگــى نـامـيــدن نـادرسـت اسـت ، چــون الـفـبـا بـاخـتـرى يـا خـاورى نـيـسـت . الـفـبـا فـرمـانـبـردار هـيـچ كـشــورى نـيـسـت و هـيـچـگــونـه ارزش تـاريـخــى نـدارد . تـنـهـا ابــزاريـسـت بـراى خــوانــدن و نــوشـتـن . بـراى درسـت و نـيـك انـجـام دادن هـر پـيـشـه يــى بـه ابــزارى كـارا نـيـاز اسـت . بـهـتـريـن ابــزار بـراى خــوانــدن و نــوشـتـن پـارســى الـفـبـايــى چــون پـارســى جـهـانــى اسـت .

۷. خــوشـنــويـســى و نـگـارشـهـاى زيـبـاى نـسـتـعـلـيــق و شـكـسـتـه از مـيـان خــواهـنـد رفـت .

پـاسـخ:
بـا ايـنـكـه نـسـتـعـلـيــق و شـكـسـتـه روشـهـاى زيـبـايــى بـراى خـشـنــويـســى هـسـتـنـد ، امــروزه بـسـيـار كـم بـه كـار مـيــرونـد . بـه جـاى ايـشـان ، كـلـكـهـاى نـازيـبـايــى چــون نـسـخ هـرچـه بـيـشـتـر در چـاپ و بـر روى ايـنـتــرنـت بـه كـار بـرده مـيـشـونـد . دلـيـل آن ايــن اسـت كـه بـه كـار بـردن كـلـكـهـاى زيـبـا عـمـلــى نـيـسـت . بـا وجــود ايـنـكـه دبـيــره ى كـنــونــى دبـيــره ى پــارســى-عـربــى اسـت ، وجــود نـسـتـعـلـيــق و شـكـسـتـه تـا بـه ايــن انــدازه در خـطــر نـبــوده اسـت .

هـنـر خــوشـنــويـســى مـيـتـوانـد از بــودن د و دبـيــره ى گــونـاگــون بـهــره ى فـراوان بـبـرد . خــوشـنــويـسـان خــواهـنـد تـوانـسـت كـارهـاى هـنـرى نـسـتـعـلـيـق و شـكـسـتـه ى خــود را دنـبـال كـرده و دبـيــره ى نـو را نـيــز در هـنـرشـان در بـر بـگـيــرنـد .

۸. بـايـد كـه بـه هـمـگـان دبـيــره ى نـو را بـيـامــوزانـيـم .

پـاسـخ:
درسـت اسـت . ولــى اسـتـادى در خــوانــدن و نــوشـتـن بـه دبـيــره يــى واجــى بـر پـايـه ى لـاتـيـن هـمـچــون پـارســى جـهـانــى ، بـراى كــودكـان چـنـد مـاه ، و بـراى بـاسـوادان بـيـش از چـنـد دقـيـقـه طــول نـخــواهـد كـشـيـد . اسـتـادى در الـفـبـاى پــارســى-عـربــى بـه ۹ سـال تـمــريـن هـرروزى نـيـاز دارد . بـا ايــن حـال فـراگـيــرى هـمـگــى روشـهـا و اسـتـثـنـا هـاى گــونـاگــون هـرگــز پـايـانــى نـدارنـد .

۹. انــديـشـه ى جـايـگـزيـنــى الـفـبـاى كـنــون بـا الـفـبـاى نـو كـار ايــرانـيـان بـيــرون از كـشــور اسـت . در ايــران مـردم بـه چــونـيـن مـسـالـه هـايــى نـمــى انــديـشـنـد چــون بـا مـسـالـه هـاى بـزرگـتـرى روبــرويـنـد .

پـاسـخ:
انــديـشـه ى دگـرگــونــى دبـيــره ى كـنــونــى پـارســى از خــود ايــران آغـاز شـد . الـفـبـاهـاى بـسـيـارى مـانـنـد پـارســى جـهـانــى در آنـجـا بـنـيـاد نـهـاده شـدنـد . آشـكـار نـبــودن پـشـتـيـبـانــى از الـفـبـاى نـو دلـيـلـهـاى قـانــونــى ، سـيـاســى و نـبــودن جـسـتـار و پــويـش آزاد مـيـان مـردم دارد . هـنـگـامــى كـه آزادى سـخـن و انــديـشـه بـاز گـردد ، خــواهـيـم ديـد كـه دوسـتــى و دلـبـسـتـگــى بـه الـفـبـاى نـو نـيــز بـار ديـگـر بـاز خــواهـد گـشـت .

۱۰. تـنـهـا واكـه هـاى كــوتـاه نــوشـتـه نـمـيـشـونـد ، ولــى واكـه هـاى بـلـنـد در دبـيــره ى كـنــونــى وجــود دارنـد .

پـاسـخ:
كـمـاســى واكـه هـاى كــوتـاه يـكــى از نـارسـايــى هـاى بـنـيـادى الـفـبـاى پــارســى-عـربــى اسـت . تـنـهـا وجــود ۵۰% از واكـه هـا بـراى خــوانــدن و نــوشـتـن زبـان پـارســى بـسـنــده نـيـسـت . بـر خـلـاف زبـان عـربــى ، ســرشـت زبـان پـارســى ، مـانـنـد ديـگـر زبـانـهـاى هـنـد و اروپـايــى ، نـيـازمـنـد ايــن اسـت كـه هـمـه ى واكـهـا نـمـايـان بـاشـنـد . اگـرنـه ، بـا مـسـالـه ى مـوجــود كـنــونــى روبــرو مـيـشـويـم ، يـعـنــى واژهـايــى بـا فـراگــويـشـهـاى گــونـاگــون كـه بـا يـك جــور امـلـا نــوشـتـه مـيـشـونـد ، يـا خــوانـا نـبــودن فـراگــويـشـهـاى واژه هـاى نـا آشـنـا . تـنـهـا ايــن كـمـبــود كـامـلـا دبـيــره ى كـنــونــى را بـسـتــوه در آورده اسـت .

۱۱. يـك نـمــونـه ى بـرجـسـتـه بـراى ازهـم گـسـيـخـتـگــى زبـان و الـفـبـا ، بـرنـامـه هـايــى اسـت كـه ۷۰ سـال پـيـش آتـاتـرك در تـركـيـه پـيـاده كـرد . فـرجـام ايــن بــود كـه تـركـان كـامـلـا از تـاريـخ و فـرهـنـگ پـرشـكــوه خــود ، بـه خـاطــر دگـرگــونــى الـفـبـا ، بـريــده شـدنـد . بـيـشـتـر تـركـان زيــر ۶۰ سـال نـمـيـتـوانـنـد از زيـبـايــى كـارهـاى خــوشـنــويـسـان در هـنـرهـا و سـاخـتـمـانـهـاى زمـان عـثـمـانــى بـهــره مـنـد گـردنـد يـا نــوشـتـه هـاى پـيـش از آتـاتـرك را بـخــوانـنـد .

پـاسـخ:
پـافـشـارى بـر ايـنـكـه تـركـان از تـاريـخ و فـرهـنـگ خــود بـريــده شـده انـد صـددرصـد نـادرسـت اسـت . امــروز بـيـش از پـيـش كـتـابـهـا ، نــوشـتـه هـا و آگـاهـيـهـايــى دربـاره ى فـرهـنـگ ، ادب و تـاريـخ تـركــى در دسـتــرس هـمـگـان قـرار دارد . هـمـاهـنـگ بـا ايــن ، تـركـان تـوانـسـتـه انـد در مــورد آمــوزش و پـرورش ، بـيـش از هـر زمـان در تـاريـخـشـان ، بـه رده ى والـاتـرى دسـت بـيـابـنـد . فـرهـنـگ تـركــى بـا دگـرنــونــى الـفـبـا ، شـكــوفـاتـر شـده اسـت . اديـبـان بـزرگــى چــون احـمـد هـاشـم ، نـاظـم هـتـمـت ، عـزيــز نـسـيـن و ديـگـران پـديــدار گـشـتـه انـد . بـنــدى كـه پـيـش دوسـتــداران فـراگـيــرى زبـان ، ادب و فـرهـنـگ تـركــى را بـسـتـه بــود بـر داشـتـه شـده ، و بــاعــث شـده كـه ايــن زبـان بـه گـروه زبـانـهـاى ارزشـمـنـد جـهـان بـپـيــونــدد . 

امــروز هـم از دسـتـكـارهـاى هـنـرى خــوشـنــويـسـان زمـان عـثـمـانــى مـيـتـوان هـمـانـنـد گــذشـتـه بـرخــوردار شـد . خــوشـنــويـســى هـنـريـسـت كـه بـه خـاطــر جـلــوگـيــريـهـايــى كـه ديـن اسـلـام از هـنـرهـاى نـگـارشــى و تـنــديـس سـازى كـرده بــود ، نـيــرو گــرفـتـه بــود . پـيـش از نــو انـگـيــزى مـاشـيـن چـاپ ، خــوشـنــويـســى بـيـش از يـك هـنـر زيـبـا در نـزد نـگـارنــدگـان كـتـابـهـاى خـطــى بــود . هـمـچــون ابــزارى بــود ويــژه بـراى نــوشـتـن و ويــرايـش كـتـابـهـا . از زمـان پـيــدايـش مـاشـيـنـهـاى چـاپ ، خــوشـنــويـســى ديـگـر ابــزارى لـازم بـراى كـتـابـنـگـارى نـيـسـت ، و پـايـگـاهــى شـايـسـتـه مـيـان هـنـرهـاى زيـبـاى ديـگـر يـافـتـه اسـت . بـنـابـر ايـن ، نـگـه داشـتـن دبـيــره ى كـنــونــى بـا هـمـه ى دشــواريـهـا و پـيـچـيــدگــى هـايـش ، تـنـهـا بـه خـاطــر نـگـهــدارى هـنـر خــوشـنــويـســى ، آونــدى بــى پـايـه اسـت . 

اكـنــون نـگـاهــى بـيـنــدازيـم بـه نـاتـوانــى تـركـان در خــوانــدن نــوشـتـه هـاى زمـان عـثـمـانـيـان . چـنـد پـرسـش بـجـا خــواهـنـد تـوانـسـت پـرده از روى نـهـاد ايــن مـسـالـه ى سـاخـتـگــى بـر دارنـد . چـنـد درصـد بـه كـار بـردن الـفـبـا ، روزانـه ، مـاهـانـه ، يـا سـالـانـه در كـشــورهـايــى چــون ايــران و تـركـيـه ، صــرف خــوانــدن نــوشـتـه هـاى خـطــى يـا كـهـن مـيـشـود ؟ آيـا بـهـتـر نـيـسـت كـه رونــوشـتـهـايــى از ايــن نــوشـتـه هـا بـه دبـيــره ى نـويـن بـازنــويـس شـونـد تـا هـمـگــى مـردم بـتـوانـنـد از خــوانــدن آنـان بـهــره مـنـد گـردنـد ؟ بـراى نـمــونـه مــورد ايــران را در نـظــر بـگـيــريـم چــون هـنــوز الـفـبـاى كـهـن در آن رواج دارد . و گـمـان كـنـيـم كـه در درازاى روز ايــرانـيـان زمـان و فـرصـت آن را بـيـابـنـد كـه نــوشـتـه يــى كـهـن را نـگـاهــى بـيـنــدازنـد . چـنـد درصـد ، گـمـان مـيـكـنـيـد ، كـه بـتـوانـنـد آن را بـه آسـانــى بـخــوانـنـد ؟ پـاسـخ ايــن پـرسـش آشـكـار اسـت و كـامـلـا ايــن مـسـالـه ى خــوانــدن نــوشـتـه هـاى خـطــى و كـهـن را از مـيـان بـرمـيــدارد . گــذشـتـه از هـمـه چـيــز ، پـژوهـش در نــوشـتـه هـاى كـهـن ويــژه ى زبـانـشـنـاسـان و تـاريـخــدانـان اسـت نـه هـر كـســى . پـژوهـشـگـرانــى كـه گـسـتـره ى كـارشـان بـايـسـتـه ى آن اسـت كـه كـتـابـهـاى كـهـن را بـررســى كـنـنـد ، بـايـد آشـنـايــى كـامـل بـا خــوانــدن و نــوشـتـن بـه زبـان ايــن نــوشـتـه هـا داشـتـه بـاشـنـد .

۱۲. مـسـالـه روش آمــوزش خــوانــدن و نــوشـتـن اسـت نـه دبـيــره . دبـيــره ى نــو نــوشــداروى هـمـه ى دردهـا نـيـسـت .

پـاسـخ:
هـرگــز چـيــزى نــوشــدارو نـمـيـشـود . بـا ايــن حـال ، پـاره يــى از راه حـلـهـا بـراى دشــواريـهـا كـاراتـر از پـاره يــى ديـگـرنـد . راه حـلــى كـارا بـراى بـالـا بـردن پـايـه ى بـاسـوادى پـارســى زبـانـان ، و دسـتــرســى دادن جـهـانـيـان بـه زبـان پـارســى ، گـزيـنـش الـفـبـايــى واجــى بـر پـايـه ى لـاتـيـن هـمـچــون الـفـبـاى پـارســى جـهـانــى مـيـبـاشـد . آيـا ايــن الـفـبـاى نــو دارويــى بـراى درمـان هـمـه ى دردهـاسـت ؟ تـنـهـا پـاسـخ ايـن اسـت كـه چــونـيـن دارويــى هـرگــز يـافـتـه شـدنــى نـيـسـت . در جـهـان پـيـشــرو امــروز ، بـه كـار بـردن راه حـلـهـاى كـهـنـه ى گــذشـتـه بـراى حـل دشــواريـهـاى تـازه ، و امـيـد داشـتـن بـه ايـنـكـه ايــن راه حـلـهـا فـرجـامـهـايــى نــو بـه بـار بـيـاورنـد ، چـيــزى جـز ديــوانـگــى نـيـسـت . راه حـلـهـاى تـازه بـراى آمــوزش دبـيــره ى پــارســى-عـربــى هـيـچـگـاه يـك چـيــز را دگـرگــون نـخــواهـنـد كـرد ، و آن ســرشـت و نـهـاد درهـم پـيـچـيــده ، درهـم آمـيـخـتـه ، و نـارسـاى ايــن دبـيــره اسـت . هـيـچـگــونـه چـاره سـازى يـا ويــرايـشــى تـوانـايــى ايــن را نـدارد كـه الـفـبـاى پــارســى-عـربــى را تـبــديـل بـه الـفـبـايــى واجــى ، سـاده ، و سـازگـار بـا پـيـشــرفـتـهـاى فـنــى نـويـن بـكـنـد . بـه ايـن مـيـمـانـد كـه كـســى امـيـد داشـتـه بـاشـد هـمـان شـيــوه ى تـرابـرى كـهـن اسـب و درشـكـه بـتـوانـد پـايـاپـاى تــرن تـنــدرو امــروزى پـيـش بــرود . در جـهـان امــروز بـا شـگــردهـاى نــو كـنــونــى ، تـنـهـا جـاى درخــور اسـب و درشـكـه مــوزه اسـت . دربـاره ى شـيــوه ى نـگـارش پـارســى نـيــز هـمـچــونـيـن دلـيـلــى بـســزاسـت .

۱۳ . امـلـاى انـگـلـيـســى نـيــز بـسـيـار پـيـچـيــده اسـت . بـسـيـارى از واژه هـاى هـمـفـراگــويـش را مـيـشـود چـنـد جــور نــوشـت . مـانـنـد . ايـن پـيـچـيــدگـيـهـا بــاعــث پـس مـانــدن انـگـلـسـتـان يـا آمــريـكـا در صـنـعـت و دمـكــراســى نـشـده اسـت . زبـان انـگـلـيـســى در هـمـه ى جـهـان رواج يـافـتـه اسـت .

پـاسـخ:
پـيـچـيــدگــى امـلـاى انـگـلـيـســى بـه خـاطــر واجــى نـبــودنـش اسـت . انـگـيــزه ى امـلـاى كـنــونــى هـمـيـشـه نـگـهــدارى چـهــره ى تـاريـخــى واژه هـا بـه دلـيـل ريـشـه شـنـاســى بــوده اسـت . بــويــژه در مــورد واژه هـاى بـيـگـانـه ى لـاتـيـن ، يــونـانــى ، و فـرانـسـوى . بـنـابـر ايـن ، دشــوارى هـاى سـاخـتـارى امـلـاى انـگـلـيـســى ، تـنـهـا ويــژه ى آن زبـانـنـد و نـه دبـيــره هـاى لـاتـيـنــى ديـگـر زبـانـهـا . بـراى نـمــونـه مـيـتـوان دبـيــره هـاى لـاتـيـنــى آلـمـانــى و اسـپـانـيـلــى را گـفـت كـه بـسـيـار نـزديـك بـه واجــى بــوده و بـراى هـمـيـن آسـان مـيـتـوان آمــوخـتـشـان . دبـيــره ى پـارســى جـهـانــى از ايـشـان نـيــز سـاده تـر مـيـبـاشـد چــون كـامـلـا سـاخـتـارى واجــى دارد .

سـخـتـيـهـا و نـداشـتـن روشــى بـراى امـلـاى انـگـلـيـســى ، مـسـالـه ايـسـت كـه سـالـهـاسـت در آن كـار شـده و مـيـشـود . امــروز كــوشـشـهـاى فـراوانــى از ســوى تــوده ى مـردم بـراى پـيــرايـش امـلـاى انـگـلـيـســى در انـجـام اسـت . فـرق مـيـان دبـيــره ى انـگـلـيـســى و پــارســى-عـربــى ايـن اسـت كـه انـگـلـيـســى را مـيـتـوان پـيــراسـت ولــى پــارســى-عـربــى را نـمـيـتـوان . بـر خـلـاف پــارســى-عـربــى ، الـفـبـاى انـگـلـيـســى چــون بـر پـايـه ى لـاتـيـن اسـت ، تـوانـسـتـه در دسـتــرس هـمـه ى جـهـانـيـان بـاشـد ، بــدون ايـنـكـه نـيـازى بـه يـادگـيــرى الـفـبـاى نـا آشـنـا بـاشـد .

پـاسـخ بـه آن آونـد كـه سـخـتــى امـلـاى انـگـلـيـســى جـلـوى پـيـشــرفـت چـشـمـگـيــر انـگـلـسـتـان يـا آمــريـكـا را نـگــرفـتـه ، درسـت مـانـنـد پـاسـخ در بـاره ى دبـيــره ى ژاپـنــى مـيـبـاشـد . بـخـشـهـاى پـهـنـاورى از گـيـتــى را مـيـتـوان انـگـلـيـســى زبـان دسـتـه بـنــدى كـرد . ايــن بــاعــث شـده كـه انـگـلـيـســى بـيـگـمـان زبـان جـهـانــى بـازرگـانــى ، عـلـم ، و شـگــرد بـشـمـار بـيـايـد . پـس هـيـچ جــور نـبـايـد نـيــرو و تـوانـايــى كـشــورى چــون ايــران را بـا تـوانـايــى كـشــورهـاى انـگـلـيـســى زبـان سـنـجـيـد . چــرا بـا نـگـهــدارى الـفـبـاى بـر افـتـاده ى پــارســى-عـربــى ، پـارســى زبـانـان را از فـرا گــرفـتـن دانـشــى والـاتـر و ديـگـران را از يـادگـيــرى زبـان پـارســى بـاز داريـم ؟ سـنـجـش دبـيــره ى پــارســى-عـربــى بـر پـايـه ى دبـيــره هـاى ژاپـنــى و انـگـلـيـســى تـنـهـا بـه خـاطــر پـا بـر جـاى نـگـاه داشـتـن آن دبـيــره اسـت . بـا پـا بـر جـاى نـگـاه داشـتـن پــارســى-عـربــى تـنـهـا كـار انـجـام شـده جـابـجـا كـردن بـار دشــواريـهـاى الـفـبـاى كـنــونــى بـر دوش آيـنــدگـان اسـت .

۱۴. نــوشـتـن بـه پــارســى-عـربــى بـسـيـار تـنــدتـر از نــوشـتـن زبـان پـارســى بـه دبـيــره ى لـاتـيـن اسـت .

پـاسـخ:
ايـن آونـد را بـايـد نـاديــده گــرفـت . ارونــدى انــدازه يــى نـسـبــى اسـت . امــروزه بـيـشـتـر نــوشـتـه هـا بـه چـهــره ى الـكـتــرنـيـك هـسـتـنـد كـه بـا دسـت تـنـد نــوشـتـن را بـر انــداخـتـه اسـت . ايــن روزهـا ديـگـر كـســى ارزشــى بـراى تـنـد نــوشـتـن قـائـل نـمـيـشـود . ولــى بـراى آرامـش بـخـشـيــدن بـه كـسـانــى كـه ايــن نـكـتـه بـرايـشـان ارزشـمـنـد اسـت بـايـد گـفـت كـه روش تـنــدنــويـســى لـاتـيـن زمـان درازيـسـت كـه مـوجــود مـيـبـاشـد .

۱۵. زبـان و دبـيــره ى عـربـــى جـزئـى از فـرهـنـگ ايــران و زبـان پـارسـيـسـت .

پـاسـخ:
يـكـبـار ديـگـر بـايـد يـاد آورى كـرد كـه زبـان و الـفـبـا د و چـيــز جـداگـانـه انـد . وامــواژگـان عـربـــى در زبـان پـارســى را نـيــز مـيـتـوان بـه آسـانــى بـه الـفـبـاى پـارســى جـهـانــى نــوشـت . بـرنـامـه ى زدايـش وامــواژگـان از زبـان پـارســى را بـايـد از بـرنـامـه ى شـنـاسـايــى الـفـبـاى نـو بـاز شـنـاخـت . هـمـانـگــونـه كـه چـنـد بـار گــوشــزد شـد ، الـفـبـا تـنـهـا ابــزاريـسـت بـراى خــوانــدن و نــوشـتـن ، و هـيـچـگــونـه ارزش فـرهـنـگـيــى نـدارد . ولــى زبـان چــونـيـن ارزشــى را داراسـت . دگـرگــونــى الـفـبـا در فـرهـنـگ و زبـان هـيـچـگــونـه نـشـانــى نـمـيـگــذارد . هـرچـنـد ، الـفـبـا بـه گــونـه ى نـاراسـتـا مـيـتـوانـد زبـان را پـشـتـيـبـانــى و زنــده كـرده يـا آن را از تـوانـايــى انــداخـتـه و خــوار كـنـد . بـراى هـمـيـن اسـت كـه الـفـبـايــى را بـايـد گـزيـد كـه يـاور و تـوانـگـر زبـان بـاشـد . چــونـيـن الـفـبـايــى بـايـد بـه گــونـه يــى ويــراسـتـه گـردد كـه زاور زبـان و هـمـه ى نـيـازهـاى آن بـاشـد . پـارســى جـهـانــى درسـت يـكـچــونـيـن الـفـبـايـيـسـت .

۱۶. بـرابـر اصـل ۱۵ قـانــون اسـاســى جـمـهــورى اسـلـامــى ايــران ، پــارســى-عـربــى الـفـبـاى رسـمــى ايــران بـشـمـار مـيـايـد . دولـت ايــران بـيـگـمـان بـا الـفـبـاى نـو دشـمـنــى كـرده و آن را نـاروا خــواهـد دانـسـت .

پـاسـخ:
تـنـهـا چـيــزى كـه مـيـتـوان دربـاره ى ايــن آونـد گـفـت ايــن اسـت كـه قـانــون را مـيـشـود عـوض كـرد . قـانــونــى كـه امــروز اسـتـوار بـه چـشـم مـيـايـد ، فـردا نـگــونـسـار خــواهـد شـد . وجــود چــونـيـن قـانــونـهـايــى هـرگــز از دشــواريـهـاى سـتـرگ سـاخـتـارى الـفـبـاى كـنــونــى پـارســى نـخــواهـنـد كـاسـت . تـنـهـا ابــزار كـارگـر ، كـنـشــى كـارا بـا هـدفــى روشـن و آشـكـار اسـت . زمـانــى كـه مـردم بـه چـشـم خــود بـهــره ى بـه كـار بـردن الـفـبـاى پــارســى جـهـانــى را بـبـيـنـنـد ، هـرگــونـه بـنــدى بـر سـر راه گـزيـنـش آن را بـر خــواهـنـد داشـت .

۱۷. ابـرنـيــروهـاى بـيـگـانـه خــواهـان جـايـگـزيـنــى الـفـبـاى خــود بـراى الـفـبـاى اسـلـامــى هـسـتـنـد. از مـيـان بـردن الـفـبـاى عـربـــى جـزء بـرنـامـه هـاى پـنـهـانــى و زمـيـنـه سـازيـهـاى بـزرگــى اسـت كـه انـگـيــزه اشـان نـابــود كـردن راه و روش اسـلـامــى اسـت .

پـاسـخ:
ايـن آونـد صـددرصـد نـادرسـت اسـت . هـيـچـگــونـه زمـيـنـه سـازى پـنـهـانـيــى از ســوى كـشــورهـاى بـاخـتـرزمـيـن در كـار نـيـسـت . چــونـيـن پـاسـخــى هـرگــز سـازنــدگـان ايـن افـسـانـه هـا را خـرسـنـد نـخــواهـد تـوانـسـت بـكـنـد . دريـغـا كـه هـيـچ چـيــزى نـمـيـتـوان گـفـت و كـارى نـمـيـتـوان كـرد تـا بـه ايــن افـسـانـه سـازان فـهـمـانـد كـه ايــن كــوشـشـهـا هـمـه از ســوى مـردمــى اسـت كـه زبـان پـارســى را دوسـت داشـتـه و آن را گــرامــى مـيــدارنـد .

پـارســى جـهـانــى الـفـبـايــى اسـت كـه بـراى جـايـگـزيـنــى الـفـبـاى پــارســى-عـربــى ويــراسـتـه شـده ، نـه الـفـبـاى عـربـــى . زبـان عـربــى هـمـچــونـان بـه الـفـبـاى عـربــى ، كـه بـراى آن ويــراسـتـه گـرديــده ، نــوشـتـه خــواهـد شـد . بـنـابـر ايـن ، هـيـچـيـك از راه و روشـهـاى اسـلـامــى ، بـا بـرگـزيــدن الـفـبـاى نــو ، دگـرگــون نـخــواهـنـد شـد .

۱۸. در هـمـه ى سـرزمـيـنـهـاى اسـلـامــى ، يـك الـفـبـا فـراگـيــر اسـت كـه هـمـه ى كـارهـاى تـاريـخــى ، فـرهـنـگــى ، مـنـشــى ، قـانــونــى ، انـجـمـنــى ، ادبــى ، و هـنـرى بـه آن نــوشـتـه شـده و نـگـهــدارى مـيـشـونـد . زبـانـهـاى هـمـگــى ايــن كـشــورهـاى اسـلـامــى بـر پـايـه ى زبـان قــرآن كـه عـربــى بـاشـد ، بـنـيـاد نـهـاده شـده انـد .

پـاسـخ:
آونـد بـالـا كـامـلـا نـادرسـت اسـت . چـنــديـن كـشــور اسـلـامــى الـفـبـاى لـاتـيـن را بـه كـار مـيـبـرنـد . بـراى نـمــونـه مـيـتـوان تـركـيـه و انــدونــزى را نـامـيـد . آيـا چــونـيـن چـيــزى از اسـلـامــى بــودن تـركـان و انــدونــزيـان كـاسـتـه اسـت ؟ پـاسـخ چـشـمـگـيــر ايـن پـرسـش ، نـه مـيـبـاشـد . بـاز بـايـد گــوشــزد كـرد كـه زبـان و نــوشـتـن د و چـيــزنـد . زبـان پـارســى پـيــوسـتـه بـه خـانــواده ى زبـانـهـاى هـنـد و اروپـايــى اسـت ولــى بـه دبـيــره ى عـربــى نــوشـتـه مـيـشـود . زبـان تـركــى از خـانــواده ى زبـانـهـاى تـركــى مـغــولــى اسـت ، ولــى بـه دبـيــره ى لـاتـيـن نـگـاشـتـه مـيـگـردد . نـمــونـه هـاى فـراوانــى از كـشــورهـاى اسـلـامــى كـه الـفـبـاهـاى گــونـاگــونــى را بـه كـار مـيـبـرنـد وجــود دارد .

۱۹. هـمـگــى دانـشـمـنــدان اسـلـامــى ، كـه شـامـل ايــرانـيـان نـيــز مـيـشـود ، بـه زبـان و دبـيــره ى عـربــى نــوشـتـه انـد .

پـاسـخ:
آمـارى كـه ايـن گــذار را اسـتـوار كـنـد وجــود نـدارد . خــواهـشـمـنــديـم ، اگـر آن را يـافـتـيـد ، بـرايـمـان بـفــرسـتـيـد . فـرزانـگـان و اسـتـادان اسـلـامــى و رشـتـه هـاى وابـسـتـه بــدان نـاچـارنـد كـه بـا زبـان عـربــى و الـفـبـاى آن آشـنـايــى كـامـل داشـتـه بـاشـنـد . هـيـچـكـس نـبـايـد بـه آن چـشـم داشـتـه بـاشـد كـه هـمـه ى مـردمـان كـاتــولـيـك بـا زبـان و دبـيــره ى لـاتـيـن آشـنـايــى داشـتـه بـاشـنـد . دربـاره ى مـسـلـمـانـان نـيــز هـمـيـن را بـايـد انـگـاشـت . چــون مـردمـانــى بـه ديـن اسـلـام بـاور دارنـد نـاچـار نـيـسـتـنـد كـه زبـان و دبـيــره ى عـربــى را فـرا گــرفـتـه بـاشـنـد . بـاز هـم بـايـد گــوشــزد كـرد كـه تـركـان و انــدونــزيـان بـا ايـنـكـه الـفـبـا و زبـان عـربــى را نـمـيــدانـنـد ، مـانـنـد بـيـشـتـر عـربــزبـانـان مـسـلـمـان شـنـاخـتـه مـيـشـونـد .

۲۰. بـر كـنـار كـردن نــويـسـه هـاى ويــژه ى زبـان عـربـــى بــاعــث فـراگــويـش نـادرسـت خــواهـد شـد .

پـاسـخ:
اگـر هـدف ايــن آونـد وامــواژه هـاى عـربــى در زبـان پـارســى اسـت ، بـايـد گـفـت كـه بـسـيـارى از آنـان نـهـاد عـربـــى خــود را از دسـت داده و پـارســى شـده انـد . ايـن واژه هـا هـيـچـگـاه مـانـنـد بـن عـربــى خــود فـراگـفـتـه نـمـيـگـردنـد . نــويـسـه هـاى ويــژه ى عـربــى درسـت مـانـنـد نــويـسـه هـاى سـاده ى پـارســى فـراگـفـتـه مـيـشـونـد . پـس دربـاره ى وامــواژه هـا ايــن آونـد پـا بـر جـاى نـمـيـمـانـد .

اگـر ايـن دربـاره ى زبـان عـربــى اسـت ، بـايـد گـفـت كـه ايــن زبـان الـفـبـاى خــود را نـگـاه داشـتـه و فـرزانـگـان اسـلـامــى هـمـچــون پـيـش بـه خــوانــدن آن خــواهـنـد پـرداخـت . تـرجـمـه هـاى نــوشـتـه هـاى عـربــى نـيــز ، هـمـانـنـد امــروز ، در دسـتــرس كـسـانــى كـه عـربـيــزبـان نـيـسـتـنـد قـرار خــواهـد داشـت .

۲۱. آيـا ايــران سـامـان بـهـتـرى مـيــداشـت اگـر رضـا شـاه الـفـبـاى نـو را جـايـگـزيـن پــارســى-عـربــى مـيـكـرد؟

پـاسـخ:
نـه اگـر شـيــوه ى دگـرگــون كـردن يـكـشـبـه ى آتـاتـرك را بـر مـيـگـزيـد . دگـرگــونــى كـم كـم و آهـسـتـه از دبـيــره ى كـنــونــى بـه لـاتـيـن مـيـتـوانـسـت مـايـه ى پـيـشــرفـتـهـاى چـشـمـگـيــر درونــى و بـرونـمـرزى در ايــران بـشـود .

۲۲. تـاجـيـكـيـان كـه الـفـبـاى سـيــريـلـيـك را بـه كـار مـيـبـرنـد ، هـيـچـگـاه تـوانـايــى خــوانــدن چـكـامـه هـاى حـافـظ يـا نــوشـتـه هـاى ديـگـران را بـه دبـيــره ى پــارســى-عـربــى نـدارنـد . پـس از گـزيـنـش الـفـبـاى لـاتـيـن ، ايــرانـيـان نـيــز هـمـيـن مـسـالـه را خــواهـنـد داشـت .

پـاسـخ:
ايـن مـسـالـه تـنـهـا هـنـگـامــى رخ خــواهـد داد كـه الـفـبـا يـكـشـبـه عـوض شـود . ولــى اگـر آهـسـتـه انـجـام بـگـيــرد ، از ايــن پـيـشـامـد جـلــوگـيــرى خــواهـد شـد . تـاجـيـكـيـان بـسـيـار اسـتـادانـه حـافـظ را بـه دبـيــره ى سـيــريـلـيـك خــودشـان مـيـخــوانـنـد .
